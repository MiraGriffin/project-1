#Function for getting urban population for a country:

get_urban_pop <- function(country) {
  base_url <- "https://api.api-ninjas.com/v1/country"
  my_key <- "Y1uFEyHRSSkY4CU7PAkQfmHI8VsM3mvv75xLtbM4"
  my_url <- glue("{base_url}?name={country};X-Api-Key={my_key}")
  country_data <- fromJSON(my_url)
  
  country_urban_pop <- country_data |>
  select(urban_population) |>
  mutate(iso3 = country)
  
  return(country_urban_pop)
}
